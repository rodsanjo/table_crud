10:  * @since 20130130
11:  */
12: class Autoloader {
13:     
14:     static $depuracion = false;

17:         if (self::$depuracion) {
18:             echo "<hr />";

26:      * Esta es la función que tiene la "inteligencia" para buscar los ficheros por las carpetas del disco del servidor
27:      * @param string $class_name
28:      * @return boolean
29:      * @throws \Exception
30:      */
31:     function autoload($class_name) {

34:             echo "<br /><hr />";

40:             return;

42:         // Sustituir las \ que separan el namespaces del nombre de la clase por DS que separa carpetas
43:         //$class_name = str_replace(array("\\", ), array(DS , ), $class_name);  //Antiguo
44:                 $class_name = str_replace(array("\\"), array(DS), $class_name);

69:         }
