49: class Table_Row_Group_Frame_Decorator extends Frame_Decorator {
50: 
51:   /**
52:    * Class constructor
53:    *
54:    * @param Frame $frame   Frame to decorate
55:    * @param DOMPDF $dompdf Current dompdf instance
56:    */
57:   function __construct(Frame $frame, DOMPDF $dompdf) {
58:     parent::__construct($frame, $dompdf);
59:   }
60: 
61:   /**
62:    * Override split() to remove all child rows and this element from the cellmap
63:    *
64:    * @param Frame $child
65:    */
66:   function split($child = null) {
67: 
68:     if ( is_null($child) ) {
69:       parent::split();
70:       return;
71:     }
72: 
73: 
74:     // Remove child & all subsequent rows from the cellmap
75:     $cellmap = $this->get_parent()->get_cellmap();
76:     $iter = $child;
77: 
78:     while ( $iter ) {
79:       $cellmap->remove_row($iter);
80:       $iter = $iter->get_next_sibling();
81:     }
82: 
83:     // If we are splitting at the first child remove the
84:     // table-row-group from the cellmap as well
85:     if ( $child === $this->get_first_child() ) {
86:       $cellmap->remove_row_group($this);
87:       parent::split();
88:       return;
89:     }
90:     
91:     $cellmap->update_row_group($this, $child->get_prev_sibling());
92:     parent::split($child);
93:     
94:   }
95: }
